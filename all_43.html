OC plist文件管理(代码方式)

//获取本地沙盒路径
NSArray *path = NSSearchPathForDirectoriesInDomains(NSDocumentDirectory, NSUserDomainMask, YES);

//获取完整路径
NSString *documentsPath = [path objectAtIndex:0];
NSString *plistPath = [documentsPath stringByAppendingPathComponent:@"Test.plist"];

//创建数据
NSMutableDictionary *dataDict = [NSMutableDictionary dictionary];

[dataDict setObject:@"zhangd" forKey:@"name"];
[dataDict setObject:@"26" forKey:@"age"];
[dataDict setObject:@"man" forKey:@"sex"];

//写入plist文件
[dataDict writeToFile:plistPath atomically:YES];

//读取plist文件
//文件是什么类型，就用什么类型的数据来接收
NSMutableDictionary *dataDict1 = [[NSMutableDictionary alloc] initWithContentsOfFile:plistPath];
NSLog((@"**start**\n[function name:%s]\n" "[Lines:%d]\n" "data->%@\n**end**\n\n"), __FUNCTION__, __LINE__,dataDict1);
