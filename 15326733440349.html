获取系统信息 下面是屏幕宽高
 wx.getSystemInfo({
      success: function(res) {
        that.setData({
          windowHeight: res.windowHeight,
          windowWidth: res.windowWidth,
        })
      }
});

Toast提示
 mask   Boolean     否   是否显示透明蒙层，防止触摸穿透，默认：false
 icon   String  否   图标，有效值 "success", "loading", "none"
 
 wx.showToast({
        title: '请输入提示语',
        icon: 'none',
        mask:true,
        duration: 2000
      })

      title: '正在加载',

 setTimeout(function() {
   wx.hideLoading()
 }, 1000)
请求数据
getList: function(index) {
    var this_ = this;

    wx.showLoading({
      title: '正在加载',
      mask: true,
    })
    //发起网络请求 
    wx.request({
      url: getApp().data.urladd + "master/groups/query",
      method: "POST",
      header: {
        'content-type': 'application/x-www-form-urlencoded',
        'Authorization': 'Bearer ' + token
      },
      data: {
        page: index,
        page_size: this_.data.pageSize,
        status: this_.data.status,
      },
      success: function(e) {
        console.log(e.data.data.data);
      },
      complete: function(res) {
        wx.hideLoading()
      }
    })
  },

点击事件
bindtap  事件绑定不会阻止冒泡事件向上冒泡
catchtap  事件绑定可以阻止冒泡事件向上冒泡

点击事件的参数传递：
target：触发事件的源组件
currentTarget：事件绑定的当前事件
